description: "Instructions on how to set up each tag."
---

<ParamField query="Title">
  <Card horizontal>
    You have to find this yourself, could also be the name of the song file.
  </Card>
</ParamField>

<ParamField query="Artist">
  <Card horizontal>
    Go to the album you want to add this song to -\> click on the 3 dots -\> `properties` -\> copy the text under `artist`
  </Card>
</ParamField>

<ParamField query="Album">
  <Card horizontal>
    Go to the album you want to add this song to -\> click on the 3 dots -\> `properties` -\> copy the text under `album`
  </Card>
</ParamField>

<ParamField query="Album Artist">
  <Card horizontal>
    Go to the album you want to add this song to -\> click on the 3 dots -\> `properties` -\> copy the text under `album artist`
  </Card>
</ParamField>

<ParamField query="Year">
  <Card horizontal>
    Go to the album you want to add this song to -\> click on the 3 dots -\> `properties` -\> copy the number under `year`
  </Card>
</ParamField>

<ParamField query="Track">
  <Card horizontal>
    Go to the album you want to add this song to -\> click on the 3 dots -\> `properties` -\> take the number after `track-  of` and add 1 to it
  </Card>
</ParamField>

<ParamField query="Total Tracks">
  <Card horizontal>
    Go to the album you want to add this song to -\> click on the 3 dots -\> `properties` -\> copy the number after `track-  of` without modifying it
  </Card>
</ParamField>

<ParamField query="Discnumber">
  <Card horizontal>
    Go to the album you want to add this song to -\>

    if the album is sectioned then chose the section number you want to put it in

    if the album is not sectioned put 1
  </Card>
</ParamField>

<ParamField query="Total Discs">
  <Card horizontal>
    Go to the album you want to add this song to -\>

    if the album is sectioned then put the number of sections

    if the album is not sectioned put 1
  </Card>
</ParamField>

<ParamField query="Genre">
  <Card horizontal>
    Go to the album you want to add this song to -\> click on the 3 dots -\> `properties` -\> memerize the text under `genre` -\> go back to mp3tag and write it **_exactly_** how it was in apple music
  </Card>
</ParamField>

<ParamField query="Composer">
  <Card horizontal>
    You have to add this yourself, just put the real name of the artist/s, producer/s, writer/s, each one seperated with `,`
  </Card>
</ParamField>

<ParamField query="Date Modified">
  <Card horizontal>
    Go to the album you want to add this song to -\> go to the last song -\> right click on it -\> `properties` -\> go to the `file` tab -\> copy the text after `date modified`
  </Card>
</ParamField>

<ParamField query="Date Added">
  <Card horizontal>
    Go to the album you want to add this song to -\> go to the last song -\> right click on it -\> `properties` -\> go to the `file` tab -\> copy the text after `date added`
  </Card>
</ParamField>

<ParamField query="Copyright">
  <Card horizontal>
    Go to the album you want to add this song to -\> go to the last song -\> right click on it -\> `properties` -\> go to the `file` tab -\> copy the text after `copyright`
  </Card>
</ParamField>

<ParamField query="Bitrate">
  <Card horizontal>
    Put the bit rate of your song, can be found by clicking on `properties` on the file in file explorer
  </Card>
</ParamField>

<ParamField query="AUDIO FORMATS">
<Card horizontal>
  Must be `Dolby Atmos, Lossless, AAC`
</Card>
</ParamField>

<ParamField query="Cloud Status">
  <Card horizontal>
    Must be `Apple Music`
  </Card>
</ParamField>

<ParamField query="Location">
  <Card horizontal>
    Must be `Cloud`
  </Card>
</ParamField>

<ParamField query="Kind">
  <Card horizontal>
    Must be `Apple Music AAC audio file`
  </Card>
</ParamField>

<ParamField query="ITUNESADVISORY">
  <Card horizontal>
    Put

    0 if your song is clean\
    1 if your song is explicit
  </Card>
</ParamField>
