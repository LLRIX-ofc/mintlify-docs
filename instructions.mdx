<ParamField query="Title">
  <Card horizontal>
    You have to find this yourself, could also be the name of the song file.
  </Card>
</ParamField>

<ParamField query="Artist">
  <Card horizontal>
    Go to the album you want to add this song to -\> click on the 3 dots -\> `properties` -\> copy the text under `artist`
  </Card>
</ParamField>

<ParamField query="Album">
  <Card horizontal >
    Go to the album you want to add this song to -\> click on the 3 dots -\> `properties` -\> copy the text under `album`
  </Card>
</ParamField>

<ParamField query="Album Artist">
  <Card horizontal >
    Go to the album you want to add this song to -\> click on the 3 dots -\> `properties` -\> copy the text under `album artist`
  </Card>
</ParamField>

<ParamField query="Year">
  <Card horizontal >
    Go to the album you want to add this song to -\> click on the 3 dots -\> `properties` -\> copy the number under `year`
  </Card>
</ParamField>

<ParamField query="Track">
  <Card horizontal >
    Go to the album you want to add this song to -\> click on the 3 dots -\> `properties` -\> take the number after `track-  of` and add 1 to it
  </Card>
</ParamField>

<ParamField query="Total Tracks">
  <Card horizontal >
    Go to the album you want to add this song to -\> click on the 3 dots -\> `properties` -\> copy the number after `track-  of` without modifying it
  </Card>
</ParamField>

<ParamField query="Discnumber">
  <Card horizontal >
    Go to the album you want to add this song to -\> click on the 3 dots -\> `properties` -\> take the number after `track-  of` and add 1 to it
  </Card>
</ParamField>
